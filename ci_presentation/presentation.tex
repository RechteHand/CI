\begin{frame}{Genetischer Algorithmus: Grundidee des Inselmodells}
\begin{columns}[T]
\footnotesize
\begin{column}{0.5\textwidth}
\textbf{Konzept der unabhängigen Inseln}
\begin{itemize}
    \item Mehrere unabhängige Populationen statt einer einzigen.
    \item Jede Insel besitzt eine \textbf{eigene Kultur} (Konfiguration, Selektion, Crossover, Mutation).
    \item Parallele Suchprozesse auf mehreren CPU-Kernen (Multithreading).
\end{itemize}
\textbf{Insel-Zyklus (wiederholt):} 1. Selektion $\rightarrow$ 2. Crossover $\rightarrow$ 3. Mutation $\rightarrow$ 4. Lokale Suche $\rightarrow$ 5. Elitismus.
\end{column}

\begin{column}{0.5\textwidth}
\textbf{Warum ein Inselmodell?}
\begin{itemize}
    \item \textcolor{HFTTeal}{\textbf{Vorteile:}} Reduziert Gefahr früher Konvergenz, erhält mehrere Suchrichtungen, erhöht genetische Diversität.
\end{itemize}
\textbf{Wichtige Implementierungsentscheidung}
\begin{itemize}
    \item Jede Insel verwendet unterschiedliche Operatoren (z.\,B. Tournament vs. Roulette, PMX vs. Order Crossover).
    \item \textcolor{HFTTeal}{\textbf{Begründung:}} Es existiert kein universell bester genetischer Operator. Durch die parallele Verwendung können unterschiedliche Bereiche des Suchraums effizienter untersucht werden.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Genetischer Algorithmus: Generationswechsel}
\begin{columns}[T]
\footnotesize
\begin{column}{0.5\textwidth}
\textbf{Erzeugung einer neuen Generation}
\begin{itemize}
    \item \textbf{Ablauf:} 1. Elitismus (Bestes Individuum), 2. Elternwahl, 3. Crossover, 4. Mutation ($p_m$), 5. Lokale Suche ($p_{ls}$), 6. Fitness berechnen.
\end{itemize}
\textbf{Warum Elitismus?}
\begin{itemize}
    \item Bereits gefundene gute Lösungen gehen niemals verloren. Stabilisiert den Suchprozess.
\end{itemize}
\end{column}

\begin{column}{0.5\textwidth}
\textbf{Adaptive Mutationsrate}
\begin{itemize}
    \item \textcolor{HFTTeal}{\textbf{Auslöser:}} Sinkt die Diversität oder stagniert die Fitness, wird die Mutationsrate automatisch erhöht.
    \item \textcolor{HFTTeal}{\textbf{Ziel:}} Dynamischer Kompromiss zwischen Exploration und Exploitation.
\end{itemize}
\textbf{Lokale Suche (Memetic Algorithm)}
\begin{itemize}
    \item \textbf{Operationen:} Positionen tauschen, Segmente mischen, Neustarts, Akzeptanz schlechterer Lösungen (Simulated Annealing).
    \item \textcolor{HFTTeal}{\textbf{Begründung:}} Genetische Operatoren für globale Suche, gepaart mit lokaler Suche zur gezielten Verfeinerung vielversprechender Lösungen.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Drohnen Fußball-Show: Physik \& Kollisionsvermeidung}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\footnotesize
\textbf{1. Separation (Abstoßung)}
\begin{itemize}
    \item Drohnen stoßen sich ab, wenn sie den Mindestabstand $d$ unterschreiten.
    \item \textbf{Vektor-Formel:} $\vec{F}_{sep} = \frac{\vec{pos}_{this} - \vec{pos}_{other}}{d^2}$
    \item Die Abstoßungskraft ist umgekehrt proportional zur quadrierten Distanz.
\end{itemize}
\textbf{Spatial Partitioning (Optimierung)}
\begin{itemize}
    \item Um nicht jede Drohne mit allen anderen zu vergleichen ($O(N^2)$), wird der Raum in ein \textbf{SpatialGrid} unterteilt.
    \item Jede Drohne prüft nur ihre direkten Nachbar-Zellen $\rightarrow$ \textbf{Komplexität $O(N)$}.
\end{itemize}
\end{column}

\begin{column}{0.5\textwidth}
\footnotesize
\textbf{2. Blockaden \& Hilfsvektoren}
\begin{itemize}
    \item \textbf{Problem:} Fliegende Drohne bleibt in geparkten Drohnen stecken ($speed \approx 0$).
    \item \textbf{Notsignal:} Blockierte Drohne pulsiert rot und sendet Signal ($\vec{pos}_{broadcaster}$).
    \item \textbf{Evade-Vektor (Platz machen):}\\
    $\vec{F}_{evade} = \left(\frac{\vec{pos}_{this} - \vec{pos}_{broadcaster}}{d}\right) \times C$
    \item Geparkte Drohnen wachen auf, berechnen Fluchtvektor $\vec{F}_{evade}$ und schieben sich kurz zur Seite. Lücke entsteht!
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Blackjack Neuronales Netz: Details}
\begin{itemize}
    \item \textbf{Architektur:} Ein in C geschriebenes Feedforward Neuronales Netz ($7 \rightarrow 16 \rightarrow 2$).
    \item \textbf{Inputs (7 normierte Features):}
    \begin{itemize}
        \item Player Total, Usable Ace, Dealer Upcard, Busted Flag, Terminal Flag, Running Count, True Count.
    \end{itemize}
    \item \textbf{Aufbau der Schichten:}
    \begin{itemize}
        \item \textit{Hidden Layer:} 16 Neuronen, aktiviert durch ReLU ($max(0, x)$).
        \item \textit{Output Layer:} 2 Logits für "Hit" und "Stand", umgewandelt in Wahrscheinlichkeiten durch eine Softmax-Funktion.
    \end{itemize}
    \item \textbf{Training (Supervised Learning):}
    \begin{itemize}
        \item Berechnung des Fehlers über Cross-Entropy Loss ($L = -\log P(\text{Target})$).
        \item Backpropagation der Gradienten und Aktualisierung der Gewichte durch Stochastic Gradient Descent (SGD).
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{F1 AI Racing: Details}
\begin{itemize}
    \item \textbf{Das Konzept:} Kombination von Neuronalen Netzen mit Genetischen Algorithmen (ähnlich NEAT). Fahrzeuge erlernen die Steuerung komplett autark über Generationen hinweg.
    \item \textbf{Sensorik (16 Inputs):}
    \begin{itemize}
        \item 7 Laser-Sensoren messen Entfernungen zu Streckenbegrenzungen (Raycasting).
        \item Mathematisches Radar peilt den nächsten Checkpoint an (Distanz \& Winkel per \texttt{arctan2}).
    \end{itemize}
    \item \textbf{Fahrphysik:}
    \begin{itemize}
        \item Berücksichtigung von Reibung und Grip-Verlust bei hoher Geschwindigkeit (Untersteuern).
        \item \textit{Windschatten-Effekt (Slipstream \& Dirty Air):} Extrapush im Sog des Vordermanns, abgeschwächt in Kurven.
    \end{itemize}
    \item \textbf{Fitness \& Selektion:}
    \begin{itemize}
        \item Die besten 20\% einer Generation überleben ("Survival of the Fittest").
        \item Komplexe Fitnessbewertung (Punkte für Checkpoints, Überholen \& Überlebenszeit; massiver Abzug bei Crashes).
    \end{itemize}
\end{itemize}
\end{frame}
