\begin{frame}{Genetischer Algorithmus: Grundidee des Inselmodells}
\begin{columns}[T]
\footnotesize
\begin{column}{0.5\textwidth}
\textbf{Konzept der unabhängigen Inseln}
\begin{itemize}
    \item Mehrere unabhängige Populationen statt einer einzigen.
    \item Jede Insel besitzt eine \textbf{eigene Kultur} (Konfiguration, Selektion, Crossover, Mutation).
    \item Parallele Suchprozesse auf mehreren CPU-Kernen (Multithreading).
\end{itemize}
\textbf{Insel-Zyklus (wiederholt):} 1. Selektion $\rightarrow$ 2. Crossover $\rightarrow$ 3. Mutation $\rightarrow$ 4. Lokale Suche $\rightarrow$ 5. Elitismus.
\end{column}

\begin{column}{0.5\textwidth}
\textbf{Warum ein Inselmodell?}
\begin{itemize}
    \item \textcolor{HFTTeal}{\textbf{Vorteile:}} Reduziert Gefahr früher Konvergenz, erhält mehrere Suchrichtungen, erhöht genetische Diversität.
\end{itemize}
\textbf{Wichtige Implementierungsentscheidung}
\begin{itemize}
    \item Jede Insel verwendet unterschiedliche Operatoren (z.\,B. Tournament vs. Roulette, PMX vs. Order Crossover).
    \item \textcolor{HFTTeal}{\textbf{Begründung:}} Es existiert kein universell bester genetischer Operator. Durch die parallele Verwendung können unterschiedliche Bereiche des Suchraums effizienter untersucht werden.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Genetischer Algorithmus: Generationswechsel}
\begin{columns}[T]
\footnotesize
\begin{column}{0.5\textwidth}
\textbf{Erzeugung einer neuen Generation}
\begin{itemize}
    \item \textbf{Ablauf:} 1. Elitismus (Bestes Individuum), 2. Elternwahl, 3. Crossover, 4. Mutation ($p_m$), 5. Lokale Suche ($p_{ls}$), 6. Fitness berechnen.
\end{itemize}
\textbf{Warum Elitismus?}
\begin{itemize}
    \item Bereits gefundene gute Lösungen gehen niemals verloren. Stabilisiert den Suchprozess.
\end{itemize}
\end{column}

\begin{column}{0.5\textwidth}
\textbf{Adaptive Mutationsrate}
\begin{itemize}
    \item \textcolor{HFTTeal}{\textbf{Auslöser:}} Sinkt die Diversität oder stagniert die Fitness, wird die Mutationsrate automatisch erhöht.
    \item \textcolor{HFTTeal}{\textbf{Ziel:}} Dynamischer Kompromiss zwischen Exploration und Exploitation.
\end{itemize}
\textbf{Lokale Suche (Memetic Algorithm)}
\begin{itemize}
    \item \textbf{Operationen:} Positionen tauschen, Segmente mischen, Neustarts, Akzeptanz schlechterer Lösungen (Simulated Annealing).
    \item \textcolor{HFTTeal}{\textbf{Begründung:}} Genetische Operatoren für globale Suche, gepaart mit lokaler Suche zur gezielten Verfeinerung vielversprechender Lösungen.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Drohnen Fußball-Show: Physik \& Kollisionsvermeidung}
\footnotesize
\textbf{Startbedingungen:} Drohnen spawnen zufällig am \textbf{unteren Bildschirmrand} und erhalten dort ihr Endziel.

\vspace{0.1cm}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\textbf{1. Separation (Abstoßung)}
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}
    \item Drohnen stoßen sich ab, wenn sie zu nahe sind.
    \item \textbf{Formel:} $\vec{F}_{sep} = \frac{\vec{pos}_{this} - \vec{pos}_{other}}{d^2}$
    \item \textit{Bedeutung:} Je näher, desto extremer die Abstoßung!
\end{itemize}
\textbf{Spatial Partitioning (Optimierung)}
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}
    \item Raum wird in ein Raster (\textbf{SpatialGrid}) unterteilt.
    \item Nur direkte Nachbarn werden geprüft $\rightarrow$ extrem schnell! ($O(N)$)
\end{itemize}
\end{column}

\begin{column}{0.5\textwidth}
\textbf{2. Blockaden \& Ausweichen}
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}
    \item \textbf{Problem:} Fliegende Drohne bleibt stecken.
    \item \textbf{Lösung:} Pulsiert rot, sendet Notsignal.
    \item \textbf{Formel:} $\vec{F}_{evade} = \left(\frac{\vec{pos}_{this} - \vec{pos}_{broadcaster}}{d}\right) \times C$
    \item \textit{Bedeutung:} Geparkte weichen weg von der blockierten Drohne aus und bilden eine rettende Gasse.
\end{itemize}
\end{column}
\end{columns}

\vfill
\begin{center}
\scriptsize
\textbf{Legende:} \quad
$\vec{F}_{sep}, \vec{F}_{evade}$ = Berechnete Flug-Kräfte \quad|\quad
$\vec{pos}_{this}$ = Eigene Pos. \quad|\quad 
$\vec{pos}_{other}$ = Andere Drohne \\
$\vec{pos}_{broadcaster}$ = Blockierte Drohne \quad|\quad 
$d$ = Distanz \quad|\quad 
$C$ = Ausweich-Stärke
\end{center}
\end{frame}

\begin{frame}{F1 AI Racing: Grundidee}
\begin{itemize}
    \item \textbf{Die Simulation:} Ein vollautomatisches Formel-1-Rennen, in dem virtuelle Agenten (Autos) das Fahren komplett von Null auf erlernen.
    \item \textbf{Der Startpunkt (Anfänger):} In der ersten Generation haben die Autos keinerlei Vorwissen über Lenkung, Bremsen oder die Strecke. Sie fahren erratisch und scheiden sofort aus.
    \item \textbf{Die Entwicklung (Evolution):} Über Tausende von Generationen entwickeln sich die Agenten durch Try and Error weiter, bis sie wie Profis Ideallinien fahren und Windschatten strategisch nutzen.
\end{itemize}
\end{frame}

\begin{frame}{F1 AI Racing: Architekturwahl}
\scriptsize
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\textbf{Warum kein Q-Learning (Q-Tabelle)?}
\begin{itemize}
    \item Q-Learning erfordert diskrete Zustände (Raster).
    \item \textbf{Das F1-Problem:} 
    \begin{itemize}
        \item Exakte X/Y Koordinaten
        \item Stufenlose Sensordistanzen
        \item Fließkomma-Geschwindigkeiten
    \end{itemize}
    \item $\Rightarrow$ \textbf{State-Space Explosion!} Eine Q-Tabelle für alle diese Zustände wäre unendlich groß und würde sofort abstürzen.
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\textbf{Lösung: Neuroevolution}
\begin{itemize}
    \item Statt jeden Zustand starr zu speichern, nutzen wir ein \textbf{Neuronales Netz} als universellen Funktionsapproximator.
    \item Es lernt durch \textbf{Genetische Algorithmen} (Evolution statt Q-Values).
\end{itemize}
\end{column}
\end{columns}

\begin{frame}{F1 AI Racing: Wahrnehmung der KI}
\textbf{1. Laser Sensoren (Raycasting)}
\begin{itemize}
    \item Das Auto schießt 7 simulierte Laserstrahlen in verschiedenen Winkeln ($W$) nach vorne und zur Seite ab, um die Fahrbahnbegrenzung zu erkennen.
    \item \textbf{Formel (Ray-Wachstum in 3-Pixel-Schritten):}\\
    $\text{Step}_X = \cos(W) \cdot 3 \quad \text{und} \quad \text{Step}_Y = \sin(W) \cdot 3$
\end{itemize}

\vspace{0.3cm}
\textbf{2. Checkpoint Radar}
\begin{itemize}
    \item Das Auto orientiert sich mathematisch am nächsten Checkpoint.
    \item \textbf{Distanz (Satz des Pythagoras):}\\
    $Distanz = \sqrt{(Target_X - Car_X)^2 + (Target_Y - Car_Y)^2}$
    \item \textbf{Winkel (Trigonometrie):}\\
    $Winkel = \text{arctan2}(Target_Y - Car_Y, Target_X - Car_X) - Heading$
\end{itemize}
\end{frame}

\begin{frame}{F1 AI Racing: Evolution \& Fitness}
\textbf{Der Lernprozess (Genetischer Algorithmus)}
\begin{itemize}
    \item \textbf{Evaluation:} Die Autos fahren. Wer das Gras berührt, scheidet aus. Am Ende bekommt jedes Auto eine Fitness-Note.
    \item \textbf{Selektion (80/20-Regel):} Die schlechtesten $80\%$ sterben aus. Die besten $20\%$ (Eliten) überleben den Generationenwechsel.
    \item \textbf{Mutation:} Die Eliten werden geklont, aber ihre "Gehirn-Gewichte" werden zufällig minimal abgewandelt. So probieren die Kinder automatisch neue Manöver!
\end{itemize}

\vspace{0.2cm}
\textbf{Fitness-Design (Belohnung vs. Bestrafung):}
\begin{center}
\resizebox{\textwidth}{!}{
$Fitness = (\text{Checkpoints} \cdot 600) + (\text{Überholmanöver} \cdot 2500) + \text{PosBonus} - (\text{Zeit} \cdot 0.6) - (\text{Crashs} \cdot 2000) + (\text{Leben} \cdot 0.02)$
}
\end{center}
\begin{itemize}
    \item \textit{Strategie:} Taktisches Überholen und Fortschritt wird extrem stark gewichtet. Sinnloses Crashen wird hart bestraft, sodass dumme Fahrstile aussterben.
\end{itemize}
\end{frame}

\begin{frame}{F1 AI Racing: Realistische Fahrphysik}
Damit das Training realistisch ist, steuern die Netze ein physikalisches Modell, nicht nur einfache Pixel.

\vspace{0.3cm}
\begin{itemize}
    \item \textbf{Luftwiderstand \& Reibung:} \\
    Die Geschwindigkeit reduziert sich in jedem Frame durch Reibung, das Gaspedal muss aktiv genutzt werden.\\
    $Speed_{neu} = Speed_{alt} \cdot 0.974$
    
    \vspace{0.3cm}
    \item \textbf{Grip (Untersteuern):} \\
    Wer mit Höchstgeschwindigkeit fährt, verliert massiv an Lenkfähigkeit. Das Netz \textit{muss} lernen, vor scharfen Kurven aktiv den Fuß vom Gas zu nehmen.\\
    $Grip = 1.0 - \left( \frac{Speed}{MaxSpeed} \right) \cdot 0.32$
\end{itemize}
\end{frame}
